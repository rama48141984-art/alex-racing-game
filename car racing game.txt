    <script>
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        const scoreElement = document.getElementById('score');
        const startBtn = document.getElementById('startBtn');

        // Game Configuration
        let gameInterval;
        let isPlaying = false;
        let score = 0;
        let gameSpeed = 5;

        // Player Car
        const player = {
            x: 175,
            y: 480,
            width: 50,
            height: 90,
            speed: 7,
            color: '#ff0055'
        };

        // Enemy Cars
        let enemies = [];
        const enemyColors = ['#00ffcc', '#ffcc00', '#00ff00', '#cc00ff'];

        // Key listeners
        const keys = {
            Left: false,
            Right: false
        };

        window.addEventListener('keydown', (e) => {
            if (e.key === 'ArrowLeft') keys.Left = true;
            if (e.key === 'ArrowRight') keys.Right = true;
        });

        window.addEventListener('keyup', (e) => {
            if (e.key === 'ArrowLeft') keys.Left = false;
            if (e.key === 'ArrowRight') keys.Right = false;
        });

        function spawnEnemy() {
            // 3 lanes available: x = 50, x = 175, x = 300
            const lanes = [50, 175, 300];
            const randomLane = lanes[Math.floor(Math.random() * lanes.length)];
            const randomColor = enemyColors[Math.floor(Math.random() * enemyColors.length)];

            // Prevent stacking on top of close objects
            if (enemies.length === 0 || enemies[enemies.length - 1].y > 200) {
                enemies.push({
                    x: randomLane,
                    y: -100,
                    width: 50,
                    height: 90,
                    color: randomColor
                });
            }
        }

        function drawCar(car) {
            // Main body
            ctx.fillStyle = car.color;
            ctx.fillRect(car.x, car.y, car.width, car.height);

            // Wheels
            ctx.fillStyle = '#000';
            ctx.fillRect(car.x - 5, car.y + 10, 5, 20); // Top left
            ctx.fillRect(car.x + car.width, car.y + 10, 5, 20); // Top right
            ctx.fillRect(car.x - 5, car.y + 60, 5, 20); // Bottom left
            ctx.fillRect(car.x + car.width, car.y + 60, 5, 20); // Bottom right

            // Windshield
            ctx.fillStyle = '#e0e0e0';
            ctx.fillRect(car.x + 5, car.y + 25, car.width - 10, 15);
        }

        function drawRoad() {
            // Background grass/edges
            ctx.fillStyle = '#333';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Road Lines
            ctx.strokeStyle = '#fff';
            ctx.lineWidth = 5;
            ctx.setLineDash([20, 20]);
            
            // Lane dividers
            ctx.beginPath();
            ctx.moveTo(133, 0);
            ctx.lineTo(133, canvas.height);
            ctx.moveTo(266, 0);
            ctx.lineTo(266, canvas.height);
            ctx.stroke();
        }

        function checkCollision(rect1, rect2) {
            return rect1.x < rect2.x + rect2.width &&
                   rect1.x + rect1.width > rect2.x &&
                   rect1.y < rect2.y + rect2.height &&
                   rect1.y + rect1.height > rect2.y;
        }

        function gameOver() {
            isPlaying = false;
            clearInterval(gameInterval);
            ctx.fillStyle = 'rgba(0,0,0,0.7)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            ctx.fillStyle = '#ff0055';
            ctx.font = 'bold 30px sans-serif';
            ctx.textAlign = 'center';
            ctx.fillText('GAME OVER', canvas.width / 2, canvas.height / 2 - 20);
            
            ctx.fillStyle = '#fff';
            ctx.font = '20px sans-serif';
            ctx.fillText(`Final Score: ${score}`, canvas.width / 2, canvas.height / 2 + 20);
            
            startBtn.innerText = 'Restart Race';
        }

        function updateGame() {
            // Clear canvas
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // Draw track
            drawRoad();

            // Move player
            if (keys.Left && player.x > 10) player.x -= player.speed;
            if (keys.Right && player.x < canvas.width - player.width - 10) player.x += player.speed;

            // Draw player
            drawCar(player);

            // Manage enemies
            if (Math.random() < 0.02) spawnEnemy();

            for (let i = enemies.length - 1; i >= 0; i--) {
                enemies[i].y += gameSpeed;
                drawCar(enemies[i]);

                // Check collision
                if (checkCollision(player, enemies[i])) {
                    gameOver();
                    return;
                }

                // Remove off-screen cars & add score
                if (enemies[i].y > canvas.height) {
                    enemies.splice(i, 1);
                    score += 10;
                    scoreElement.innerText = score;

                    // Increase difficulty slowly
                    if (score % 50 === 0) gameSpeed += 0.5;
                }
            }
        }

        function startGame() {
            if (isPlaying) return;
            
            // Reset state
            isPlaying = true;
            score = 0;
            gameSpeed = 5;
            enemies = [];
            player.x = 175;
            scoreElement.innerText = score;
            startBtn.innerText = 'Racing...';

            // Game loop run at 60 FPS
            gameInterval = setInterval(updateGame, 1000 / 60);
        }

        // Initialize setup view
        drawRoad();
        drawCar(player);
        
        startBtn.addEventListener('click', startGame);
    </script>
